Tailwind + Twig and the Zed Editor

2025-06-20
:: tags:  #code  #frontend  #php  #tailwind  #twig  #zed

Getting tailwind to work in Zed for Twig was a little weird. Found this comment in an issue and thought it was a good idea to share it here.

"languages": {
  "Twig": {
    "language_servers": [
      "tailwindcss-language-server",
      "Twig Language Server"
    ]
  }
},
"lsp": {
  "tailwindcss-language-server": {
    "settings": {
      "tailwindCSS": {
        "emmetCompletions": true,
        "includeLanguages": {
          "twig": "html",
          "Twig": "html",
          "*.twig": "html",
          "plaintext": "html"
        },
        "experimental": {
          "classRegex": [
            "class=\"([^\"]*)",
            "class={\"([^\"}]*)",
            "class=format!({\"([^\"}]*)"
          ]
        }
      }
    }
  }
}
